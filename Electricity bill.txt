#include <stdio.h>
int main(){
    int units;
    printf("Enter the option \n");
    printf("1.Domestic\n");
    printf("2.Commercial\n");
    int option;
    scanf("%d",&option);
    scanf("%d",&units);

    
    
    switch(option){
        case 1:
        if(units<=100){
            units=units*3;
            printf("%d Bill ₹<amount>",units);
        }
        else{
            units=(units*3)+(units-100)*5;
            printf("%d Bill ₹<amount>",units-80);
        }
        break;
        case 2:
        if(units<=100){
            units=units*7;
            printf("%d Bill ₹<amount>",units);
        }
        else{
            units=(units*7)+(units-100)*10;
            printf("%d Bill ₹<amount>",units*10);
        }
        break;
        default:
        printf("Invalid");
    }
    return 0;
    
}
